4    class NB:
5        class_col = None
6        attribute_names = []
7    
8        prior_counts = defaultdict(int)
9        priors = defaultdict(float)
10   
11       likelihood_counts = defaultdict(float)
12       likelihoods = defaultdict(float)
13   
14       total_instances = 0
15   
16       def train(self, class_at_column=1, header=True, filepath='', delimeter=','):
17           """ 
18           Trying to create a Generalized function for training with any 
19           type of textual dataset. .csv/.tsv and with/without header in 
20           file support. 
21    
22           :param class_at_column: The column that contains the class. 
23           All other columns are considered attributes. 
24           Default class at 1st column. 
25           :param header: If the attribute names present at the first row of the file. 
26           Default is True. 
27           :param filepath: The full/relative to .py file location of data set path. 
28           :param delimeter: The attribute separator for each line. 
29           :return: None 
30           """
31           self.class_col = class_at_column - 1
32   
33           with open(filepath, mode='r', encoding='utf8') as fin:
34               """ 
35               Take the first line and based on header(True/False) 
36               Create Prior, Likelihood and Variable name lists 
37               """
38   
39               first_line = fin.readline()
40               parts = first_line.strip().split(delimeter)
41   
42               if header:
43                   i = 1
44                   for col in parts:
45                       if col in self.attribute_names:
46                           raise IOError("Column Names not Unique.")
47                       self.attribute_names.append(col.strip())
48                       i += 1
49               else:
50                   self.total_instances += 1
51   
52                   for i in range(1, len(parts) + 1, 1):
53                       self.attribute_names.append(f'column_{i}')
54   
55                   for i in range(0, len(parts), 1):
56                       if i == self.class_col:
57                           self.prior_counts[parts[i]] += 1
58                           continue
59   
60                       self.likelihood_counts[
61                           f'{self.attribute_names[i]}|{parts[i]}|{parts[self.class_col]}'] += 1
62   
63               """ 
64               Parse the full file and count all data 
65               """
66               for line in fin:
67                   self.total_instances += 1
68                   parts = line.strip().split(delimeter)
69   
70                   for i in range(0, len(parts), 1):
71                       if i == self.class_col:
72                           self.prior_counts[parts[i]] += 1
73                           continue
74   
75                       self.likelihood_counts[
76                           f'{self.attribute_names[i]}|{parts[i]}|{parts[self.class_col]}'] += 1
77   
78           for key in self.prior_counts:
79               self.priors[key] = self.prior_counts[key] / self.total_instances
80   
81           for key in self.likelihood_counts:
82               cls = key.split('|')[2]
83               self.likelihoods[key] = self.likelihood_counts[key] / self.prior_counts[cls]
84   
85       def test(self, new_instance=None):
86           if new_instance is None or len(new_instance) != (len(self.attribute_names)-1):
87               raise AttributeError(
88                   f"Invalid New Instance. Must be a list of {len(self.attribute_names)} attributes.")
89   
90           print('\n\n**********************************\n')
91           print(f'Training Instances: {self.total_instances}\n')
92           print('Prior Probabilities')
93   
94           for key in self.priors:
95               print(f'#{key}: {self.prior_counts[key]}\t\tP({key}): {self.priors[key]:.4f}')
96   
97           print('\n**********************************\n\n')
98           print(f'New instance: {str(new_instance)}\n')
99   
100          posteriors = defaultdict(float)
101  
102          for key in self.priors:
103              posteriors[key] = self.priors[key]
104  
105              i = 0
106              for col in range(0, len(self.attribute_names), 1):
107                  if col == self.class_col:
108                      continue
109  
110                  lkh = self.likelihoods[f'{self.attribute_names[col]}|{new_instance[i]}|{key}']
111                  posteriors[key] *= lkh
112                  print(f'P({new_instance[i]}|{key}): {lkh:.2f}\t', end='')
113                  i += 1
114  
115              print('')
116  
117          print('\nClass probabilities\n')
118          max_prob = 0
119          cls = ''
120          total = 0
121          for key in posteriors:
122              print(f'{key}: {posteriors[key]:.4f}')
123  
124              total += posteriors[key]
125  
126              if max_prob < posteriors[key]:
127                  max_prob = posteriors[key]
128                  cls = key
129  
130          print(f'\nClassify: {cls}')
131          print(f'Conditional Probability for class "{cls}" : {posteriors[cls]*100/total:.2f}%\n\n')
